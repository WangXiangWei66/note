  5. import  requests
  6. resp=requests.get('https://www.lingdianshuwu.com/')
  7. resp.encoding='gbk'
  8. print(resp.text)
  9. # 1. 将百度的源码保存到本地存储 (Python中文件的操作)
  10. file=open('lingdianshuwu.html','w',encoding='gbk')  # text是resp的属性  相当于 姓名是人的属性一样一样地  ,文本文件的后缀叫txt
  11. file.write(resp.text)  #text 表示的是字符串
  12. file.close()
  14. #爬虫百度的图片
  15. resp=requests.get('https://www.baidu.com/img/flexible/logo/pc/result.png')  #图片的url
  16. file=open('logo.png','wb')
  17. file.write(resp.content) # content表示的是二进制
  18. file.close()
